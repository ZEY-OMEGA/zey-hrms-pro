    <script>
        function updateDashboard(data) {
            document.querySelector("#stats .stat-card:nth-child(1) .stat-value").textContent = data.total_employees || 0;
            document.querySelector("#stats .stat-card:nth-child(2) .stat-value").textContent = data.present || 0;
            document.querySelector("#stats .stat-card:nth-child(3) .stat-value").textContent = (data.late + data.heavy_late) || 0;
            document.querySelector("#stats .stat-card:nth-child(4) .stat-value").textContent = data.absent || 0;

            const tbody = document.querySelector("#attendanceTable tbody");
            tbody.innerHTML = "";
            Object.values(data.summary || {}).forEach(emp => {
                const row = document.createElement("tr");
                row.innerHTML = `
                    <td>${emp.name}</td>
                    <td class="status-${emp.status}">${getStatusText(emp.status)}</td>
                    <td>${emp.lateness}</td>
                    <td><button onclick="notify('${emp.name}')">إرسال تنبيه</button></td>
                `;
                tbody.appendChild(row);
            });
        }

        function getStatusText(status) {
            const map = {"present": "حاضر", "late_light": "متأخر", "late_heavy": "متأخر بشدة", "absent": "غائب"};
            return map[status] || status;
        }

        function notify(name) {
            alert(`تم إرسال تنبيه إلى ${name}`);
        }

        setInterval(() => {
            fetch('/api/hr/status').then(r => r.json()).then(updateDashboard);
        }, 5000);

        // تحميل أولي
        fetch('/api/hr/status').then(r => r.json()).then(updateDashboard);
    </script>
